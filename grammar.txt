module DaKe-SYNTAX

  syntax sdfg := "SDFG" name "STARTSWITH" name "{" sdfg_body "}"
  syntax state := "STATE" name "{" state_body "}"
  syntax interstate := "INTERSTATE" "{" "FROM" name "TO" name "CONDITION" cond "EXEC" stmt "}"

  syntax state_body := access | memlet | tasklet | mapentry | mapexit
  syntax access := name ":" "READ" name |
                   name ":" "WRITE" name  

  syntax name := simple_name | quoted_name
  syntax simple_name := r"[a-zA-Z_]+[0-9a-zA-Z_]*" [token]
  syntax quoted_name := r"\"\w+\"" [token]

endmodule
